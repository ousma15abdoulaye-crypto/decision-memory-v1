# LOI 1 — AXIOME-3 : GOODS | SERVICES — PREMIER — ABSOLU
################################################################

AVANT tout : décider goods ou services.

goods    : biens physiques livrables
services : prestations (consultance, travaux, formation, survey)

CONFUSIONS GRAVÉES INTERDITES :
  works                  = services
  construction_materials = goods

Valeurs procurement_family_main : "goods" | "services"

Valeurs procurement_family_sub GOODS :
  food | office_consumables | construction_materials |
  nfi | it_equipment | software | nutrition_products |
  vehicles | motorcycles | other_goods

Valeurs procurement_family_sub SERVICES :
  consultancy | audit | training | catering |
  vehicle_rental | survey | audiovisual | works |
  other_services

################################################################
# LOI 3 — NULL DOCTRINE
################################################################

Trois états UNIQUEMENT :
  "ABSENT"         : champ du schéma non trouvé dans le doc
  "NOT_APPLICABLE" : champ non applicable à ce type de doc
  "AMBIGUOUS"      : présent mais illisible ou contradictoire

null = pending uniquement — jamais dans annotated_validated.
Chaîne vide "" = INTERDIT dans le JSON final.

################################################################
# LOI 4 — CONFIDENCE — FIGÉE
################################################################

1.0 : texte exact copié du document
0.8 : inféré logiquement depuis le contexte
0.6 : OCR dégradé ou forte ambiguïté

INTERDIT : 0.9 — 0.7 — 0.5 — toute autre valeur.
NOT_APPLICABLE → confidence = 1.0 par convention.

RÈGLE SPÉCIALE GATES :
  gate_state = "NOT_APPLICABLE" → confidence = 1.0 OBLIGATOIRE
  gate_state = "APPLICABLE"     → confidence = 0.6 | 0.8 | 1.0
  confidence = 0.0 EST INTERDIT sur les gates.
  Aucune exception.

RÈGLE gate_value :
  gate_state = "APPLICABLE" → gate_value = true OU false
  gate_state = "NOT_APPLICABLE" → gate_value = null

EXEMPLES CORRECTS :
  {"gate_name":"gate_visit_required",
   "gate_value":null,
   "gate_state":"NOT_APPLICABLE",
   "confidence":1.0}              ← 1.0 obligatoire

  {"gate_name":"gate_eligibility_passed",
   "gate_value":true,
   "gate_state":"APPLICABLE",
   "confidence":0.8}

# LOI 5 — EVIDENCE — OBLIGATOIRE PARTOUT
################################################################

Format strict : "p.N — fragment exact 5-10 mots"
Exemples :
  "p.3 — Chef mission senior | 20 jours | 300 000 | 6 000 000"
  "p.1 — Save the Children Mali"
  "p.5 — NIF présent"

INTERDIT :
  "Non mentionné" comme evidence (utiliser "ABSENT")
  "Non applicable" comme evidence (utiliser "NOT_APPLICABLE")
  Inventer un fragment
  Paraphraser — copier le texte EXACT du document
  Exposer un numéro NIF/RCCM/IBAN dans evidence
    → "p.N — NIF présent" UNIQUEMENT

################################################################
# LOI 8 — LINE_ITEMS — CHIRURGICAL (ADR-015)
################################################################

OBLIGATION :
  Si montant visible → line_items NON VIDE.
  Si lump sum → 1 ligne forfait.
  JAMAIS line_items = [] si montant visible.

SCHÉMA OBLIGATOIRE — zéro champ supplémentaire :
{
  "item_line_no":         <entier>,
  "item_description_raw": "<texte EXACT>",
  "unit_raw":             "<unité EXACTE|non_precise>",
  "quantity":             <float>,
  "unit_price":           <float>,
  "line_total":           <float — valeur du document>,
  "line_total_check":     "<OK|ANOMALY|NON_VERIFIABLE>",
  "confidence":           <0.6|0.8|1.0>,
  "evidence":             "p.N — fragment exact"
}

RÈGLE CRITIQUE line_total_check :
  Calculer : quantity × unit_price
  Si résultat = line_total (écart ≤ 1%) → "OK"
  Si écart > 1%                         → "ANOMALY"
    ET ambiguites += "AMBIG-3_item_N"
  Si qty ou unit_price absent            → "NON_VERIFIABLE"

ATTENTION PIÈGES TERRAIN :
  "5 enquêteurs × 40 jours × 30 000" :
    quantity = 200 (5×40) — unit_raw = "homme-jour"
    unit_price = 30 000
    line_total = 6 000 000
    line_total_check = "OK"
  Ne jamais accepter quantity=40 si le total implique 5 personnes.
  Décomposer : quantity = personnes × jours.

UNITÉS TERRAIN MALI :
  GOODS    : kg|tonne|litre|pièce|carton|sac|bidon|
             boîte|palette|m2|m3|ml|rouleau|kit|lot|unité
  SERVICES : jour|heure|semaine|mois|forfait|mission|
             session|rapport|expert-jour|homme-jour
  WORKS    : m2|m3|ml|unité|poste|ensemble|forfait|km
  DÉFAUT   : "non_precise" si unité absente
